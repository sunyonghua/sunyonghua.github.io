# var

var 声明变量会被提升

console.log(foo); // undefined
var foo = 1;
console.log(foo); // 1

在 var 声明变量之前访问 foo 是 undefined 这是因为当 javascript 脚本在执行时，程序会对 javascript 代码先扫描一遍,将 var 关键字声明提升到程序的最顶部

//代码变成了这样
var foo;
console.log(foo); // 此时已定义但未被赋值，输出undefined
foo = 1;
console.log(foo); //此时var被赋值为1

var 声明变量的作用域总是它当前的执行期上下文

// 全局作用域中
var baz = 'global';
console.log(baz); //global;

// 函数作用域中
function bar() {
  var foo = 1;
  console.log(foo); // 1
  console.log(baz); // global 全局作用域中用var声明的变量在任何地方都可以访问
}
bar();
console.log(foo); // foo is not defined

//（ES5 只有全局作用域和函数作用域，没有块级作用域）
if (true) {
  var a = 1;
}
console.log(a); // 1

// 从自身向上找，如果自身有的话用自身的，自身没有就一直向上找，直到找到全局window下也没有的话暂停 然后ReferenceError bar is not defined
var bar = 'is global var';
function foo() {
  console.log(bar); // undefined
  var bar = 'is function var';
}
foo();

# let

Es6 新增 声明变量只在当前块作用域内有效

// 不存在变量提升
console.log(foo); //语法错误 error
let foo = 1;

// 暂时性死区
// 只要块级作用域内声明了这个变量 它所声明的变量 就绑定了这个区域不受外部影响
let foo = 1;
if (true) {
  let foo = 2;
  console.log(foo); // 2
}

let bar = 1;
if (true) {
  console.log(bar); // 语法erroor if 内部存在 let bar 不能够访问外部 而又在声明前访问 所以报错
  let bar = 2;
}
// 总之，在代码块内，使用let命令声明变量之前，该变量都是不可用的。这在语法上，称为“暂时性死区”（temporal dead zone，简称 TDZ）。

// 不允许 重复声明

// 错误
if (true) {
  let foo = 1;
  let foo = 2;
  console.log(foo);
}
if (true) {
  let foo = 1;
  var foo = 2;
  console.log(foo);
}
if (true) {
  let foo = 1;
  const foo = 2;
  console.log(foo);
}
